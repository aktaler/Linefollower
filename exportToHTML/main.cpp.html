19   static inline void initPwmTimer(){
20       TCCR1A |= (1 << WGM11);                                             // set timer to 9-bit PWM Mode
21       TCCR1B |= (1 << WGM12) | (1 << WGM13);                              // not sure about this one
22       TCCR1B |= (1 << CS10);                                              // 1 prescaling -- counting in microseconds
23       ICR1 = 2000;                                                        // TOP value = 2ms = 500hz
24       TCCR1A |= (1 << COM1A1);                                            // when timer in pwm mode, sets not-inverting pwm signal on PB3
25       TCCR1A |= (1 << COM1B1);                                            // when timer in pwm mode, sets not-inverting pwm signal on PB2
26   }
27   static inline void initADC(){
28       ADMUX |= (1 << REFS0);                                              // reference voltage set to AVCC
29       ADCSRA |= (1 << ADPS1) | (1 << ADPS0);                              // ADC Clock prescaler = /8 = 125kHz = 8µs
30       ADCSRA |= (1 << ADEN);                                              // enable ADC
31   }
32   void initUSART(){
33       // Enable receiver and transmitter
34       UCSR0B = (1 << RXEN0) | (1 << TXEN0);
35       // Set frame format: 8 data , 1 stop bit
36       UCSR0C = (1 << UCSZ01) | (1 << UCSZ00);
37   }

40   uint16_t readADC(uint8_t channel) {
41       ADMUX = (0b11110000 & ADMUX) | channel;                             // Clear the first 4 bits of ADMUX and select ony the bits for the channel
42       ADCSRA |= (1 << ADSC);                                              // start conversion
43       while(ADCSRA & (1<<ADSC));                                          // loop till conversion is done ~ 104 µs or 13x ADC Clock speed
44       return (ADC);
45   }
46   void transmitByte(uint8_t data) {
47       // Wait for empty transmit buffer
48       do {} while (!(UCSR0A & (1 << UDRE0)));
49       // If data register empty (UDREO bit set),
50       // put data into buffer to send
51       UDR0 = data;
52   }
53   //map function for speed values
54   long MAP(long x, long in_min, long in_max, long out_min, long out_max)
55   {
56       return (x - in_min) * (out_max - out_min) / (in_max - in_min) + out_min;
57   }

74       motor2.backward();

77       uint16_t IR_left_high = 0;
78       uint16_t IR_right_low = 65535;
79       uint16_t IR_right_high = 0;
80       uint16_t current_value_left = 0;
81       uint16_t current_value_right = 0;
82   
83       while (loopValue < 1000){
84           current_value_left = readADC(IRsensor_left);
85           current_value_right = readADC(IRsensor_right);
86           // Check if ADC values are above or below currently saved extrem values
87           if (current_value_left > IR_left_high){IR_left_high = current_value_left;}
88           if (current_value_left < IR_left_low){IR_left_low = current_value_left;}
89   
90           if (current_value_right > IR_right_high){IR_right_high = current_value_right;}
91           if (current_value_right < IR_right_low){IR_right_low = current_value_right;}
92   
93           loopValue += 1;
94           _delay_ms(1);
95       }
96       loopValue = 0;
97       motor1.enable(0);
98       motor2.enable(0);
99       _delay_ms(100);
100      motor1.backward();
101      motor2.forward();
102      motor1.enable(1);
103      motor2.enable(1);
104  
105      while (loopValue < 2000){
106          current_value_left = readADC(IRsensor_left);
107          current_value_right = readADC(IRsensor_right);
108          // Check if ADC values are above or below currently saved extrem values
109          if (current_value_left > IR_left_high){IR_left_high = current_value_left;}
110          if (current_value_left < IR_left_low){IR_left_low = current_value_left;}
111  
112          if (current_value_right > IR_right_high){IR_right_high = current_value_right;}
113          if (current_value_right < IR_right_low){IR_right_low = current_value_right;}
114  
115          loopValue += 1;
116          _delay_ms(1);
117      }
118      loopValue = 0;
119      // Turn back to middle
120      motor1.enable(0);
121      motor2.enable(0);
122      _delay_ms(100);
123      motor1.forward();
124      motor2.backward();
125      motor1.enable(1);
126      motor2.enable(1);
127      _delay_ms(1200);
128      motor1.enable(0);
129      motor2.enable(0);
130      // ######################################################################### //
131      // ### PID Controller Setup ## //
132      MiniPID pid_controller_motor1(2, 1.5, 1.5);
133      MiniPID pid_controller_motor2(2, 1.5, 1.5);
134      // Calculate the range of the Setpoint values for each sensor
135      double setpointRange_left = IR_left_high - IR_left_low;
136      double setpointRange_right = IR_right_high - IR_right_low;
137      // pass the calculated range value to the object function
138      pid_controller_motor1.setSetpointRange(setpointRange_left);
139      pid_controller_motor2.setSetpointRange(setpointRange_right);
140  
141      // Pass some values to the objects
142      pid_controller_motor1.setOutputLimits(minSpeed,maxSpeed);
143      pid_controller_motor2.setOutputLimits(minSpeed,maxSpeed);
144      pid_controller_motor1.setSetpoint(IR_left_high);
145      pid_controller_motor2.setSetpoint(IR_right_high);
146      // pid_controller_motor1.setDirection(true);
147      // pid_controller_motor2.setDirection(true);
148      // set motors straight before entering the main loop
149      motor1.enable(1);
150      motor2.enable(1);
151      motor1.forward();
152      motor2.forward();
153      motor1.speed(200);
154      motor2.speed(200);
155      _delay_ms(1000);
156  
157      uint16_t cycle_reading_left;
158      uint16_t cycle_reading_right;
159      uint16_t speed_m1;
160      uint16_t speed_m2;
161      while(true){
162          cycle_reading_left = readADC(IRsensor_left);
163          cycle_reading_right = readADC(IRsensor_right);
164          speed_m1 = pid_controller_motor1.getOutput(cycle_reading_left);
165          speed_m2 = pid_controller_motor2.getOutput(cycle_reading_right);
166  
167          speed_m1 = MAP(speed_m1,minSpeed,maxSpeed,maxSpeed,minSpeed);
168          speed_m2 = MAP(speed_m2,minSpeed,maxSpeed,maxSpeed,minSpeed);
169          motor1.speed(speed_m1);
170          motor2.speed(speed_m2);
171  
172      }
